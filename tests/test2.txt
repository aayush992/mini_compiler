int count = 0;
int max = 5;
float result = 0.0;
while (count < max) {
    if (count < 2) {
        result = result + 1.5;
    } else {
        if (count < 4) {
            result = result + 2.5;
        } else {
            result = result + 3.5;
        }
    }
    count = count + 1;
}
return result;
